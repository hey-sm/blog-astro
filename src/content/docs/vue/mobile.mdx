# 移动端

Vue 可以通过 Capacitor 或其他方案构建原生移动应用。

## Capacitor

### 安装

```bash
npm install @capacitor/core @capacitor/cli
npx cap init
```

### 添加平台

```bash
npm install @capacitor/android @capacitor/ios
npx cap add android
npx cap add ios
```

### 构建与同步

```bash
npm run build
npx cap sync
npx cap open android
npx cap open ios
```

### 使用原生功能

```vue
<script setup lang="ts">
import { Camera, CameraResultType } from '@capacitor/camera'
import { ref } from 'vue'

const photo = ref<string>()

async function takePicture() {
  const image = await Camera.getPhoto({
    quality: 90,
    allowEditing: false,
    resultType: CameraResultType.Uri
  })
  
  photo.value = image.webPath
}
</script>

<template>
  <button @click="takePicture">Take Photo</button>
  <img v-if="photo" :src="photo" alt="Photo">
</template>
```

## 移动端优化

### 触摸事件

```vue
<script setup lang="ts">
import { ref } from 'vue'

const startX = ref(0)
const startY = ref(0)

function handleTouchStart(e: TouchEvent) {
  startX.value = e.touches[0].clientX
  startY.value = e.touches[0].clientY
}

function handleTouchEnd(e: TouchEvent) {
  const endX = e.changedTouches[0].clientX
  const endY = e.changedTouches[0].clientY
  
  const diffX = endX - startX.value
  const diffY = endY - startY.value
  
  if (Math.abs(diffX) > Math.abs(diffY)) {
    if (diffX > 50) {
      console.log('Swipe right')
    } else if (diffX < -50) {
      console.log('Swipe left')
    }
  }
}
</script>

<template>
  <div
    @touchstart="handleTouchStart"
    @touchend="handleTouchEnd"
  >
    Swipe me
  </div>
</template>
```

```vue
  }

</style>
```

### 性能优化

```vue
<script setup lang="ts">
import { ref, onMounted, onUnmounted } from 'vue'

// 使用 Intersection Observer 实现懒加载
const imageRef = ref<HTMLImageElement>()
const isVisible = ref(false)

onMounted(() => {
  const observer = new IntersectionObserver((entries) => {
    if (entries[0].isIntersecting) {
      isVisible.value = true
      observer.disconnect()
    }
  })
  
  if (imageRef.value) {
    observer.observe(imageRef.value)
  }
})
</script>

<template>
  <img
    ref="imageRef"
    :src="isVisible ? actualSrc : placeholder"
    alt="Lazy loaded image"
  >
</template>
```
