# 路由

Vue Router 是 Vue 官方的路由管理器，用于构建单页应用（SPA）。

## 安装与配置

### 安装

```bash
npm install vue-router@4
```

### 基础配置

```ts filename="router/index.ts"
import { createRouter, createWebHistory, type RouteRecordRaw } from 'vue-router'
import Home from '@/views/Home.vue'
import About from '@/views/About.vue'

const routes: RouteRecordRaw[] = [
  {
    path: '/',
    name: 'home',
    component: Home,
    meta: {
      title: '首页'
    }
  },
  {
    path: '/about',
    name: 'about',
    component: About,
    meta: {
      title: '关于'
    }
  },
  {
    path: '/user/:id',
    name: 'user',
    component: () => import('@/views/User.vue'),
    meta: {
      requiresAuth: true
    }
  }
]

const router = createRouter({
  history: createWebHistory(import.meta.env.BASE_URL),
  routes,
  scrollBehavior(to, from, savedPosition) {
    if (savedPosition) {
      return savedPosition
    } else {
      return { top: 0 }
    }
  }
})

export default router
```

```ts filename="main.ts"
import { createApp } from 'vue'
import App from './App.vue'
import router from './router'

const app = createApp(App)

app.use(router)

app.mount('#app')
```

## 动态路由

### 路径参数

```ts filename="router/index.ts"
const routes = [
  // 匹配 /user/123
  { path: '/user/:id', component: User },
  
  // 匹配 /user/123/post/456
  { path: '/user/:userId/post/:postId', component: Post },
  
  // 可选参数
  { path: '/user/:id?', component: User },
  
  // 匹配多个段
  { path: '/user/:id+', component: User },  // 至少一个
  { path: '/user/:id*', component: User },  // 零个或多个
  
  // 正则约束
  { path: '/user/:id(\\d+)', component: User }  // 只匹配数字
]
```

### 访问路由参数

```vue filename="views/User.vue"
<script setup lang="ts">
import { useRoute, useRouter } from 'vue-router'
import { watch } from 'vue'

const route = useRoute()
const router = useRouter()

// 访问参数
console.log(route.params.id)

// 监听参数变化
watch(
  () => route.params.id,
  (newId) => {
    console.log('User ID changed to:', newId)
    // 重新加载数据
  }
)
</script>

<template>
  <div>
    <h1>User {{ route.params.id }}</h1>
  </div>
</template>
```

### 响应路由参数变化

```vue filename="views/User.vue"
<script setup lang="ts">
import { ref, watch } from 'vue'
import { useRoute } from 'vue-router'

const route = useRoute()
const user = ref(null)

async function fetchUser(id: string) {
  const response = await fetch(`/api/users/${id}`)
  user.value = await response.json()
}

// 初始加载
fetchUser(route.params.id as string)

// 监听参数变化
watch(
  () => route.params.id,
  (newId) => {
    if (newId) {
      fetchUser(newId as string)
    }
  }
)
</script>
```

## 嵌套路由

```ts filename="router/index.ts"
const routes = [
  {
    path: '/user/:id',
```

## 编程式导航

### router.push

```vue
<script setup lang="ts">
import { useRouter } from 'vue-router'

const router = useRouter()

// 字符串路径
router.push('/about')

// 对象
router.push({ path: '/about' })

// 命名路由
router.push({ name: 'user', params: { id: 123 } })

// 带查询参数 /register?plan=private
router.push({ path: '/register', query: { plan: 'private' } })

// 带 hash /about#team
router.push({ path: '/about', hash: '#team' })

// 替换当前历史记录
router.replace({ path: '/home' })

// 前进/后退
router.go(1)   // 前进一步
router.go(-1)  // 后退一步
router.back()  // 后退
router.forward()  // 前进
</script>
```

### 导航失败处理

```vue
<script setup lang="ts">
import { useRouter, NavigationFailureType, isNavigationFailure } from 'vue-router'

const router = useRouter()

async function navigate() {
  try {
    await router.push('/about')
  } catch (error) {
    if (isNavigationFailure(error, NavigationFailureType.aborted)) {
      console.log('Navigation aborted')
    } else if (isNavigationFailure(error, NavigationFailureType.duplicated)) {
      console.log('Navigation duplicated')
    }
  }
}
</script>
```

## 路由守卫

### 全局前置守卫

```ts filename="router/index.ts"
router.beforeEach((to, from, next) => {
  // 检查认证
  if (to.meta.requiresAuth && !isAuthenticated()) {
    next({ name: 'login', query: { redirect: to.fullPath } })
  } else {
    next()
  }
})

router.beforeEach((to, from, next) => {
  // 设置页面标题
  document.title = (to.meta.title as string) || 'My App'
  next()
})
```

### 全局解析守卫

```ts
router.beforeResolve(async (to) => {
  if (to.meta.requiresCamera) {
    try {
      await askForCameraPermission()
    } catch (error) {
      return false  // 取消导航
    }
  }
})
```

### 全局后置钩子

```ts
router.afterEach((to, from, failure) => {
  if (!failure) {
    // 发送分析数据
    sendToAnalytics(to.fullPath)
  }
})
```

### 路由独享守卫

```ts
const routes = [
  {
    path: '/admin',
    component: Admin,
    beforeEnter: (to, from) => {
      if (!isAdmin()) {
        return { name: 'home' }
      }
    }
  }
]
```

### 组件内守卫

```vue
<script setup lang="ts">
import { onBeforeRouteLeave, onBeforeRouteUpdate } from 'vue-router'

// 离开当前路由前
onBeforeRouteLeave((to, from) => {
  const answer = window.confirm('确定要离开吗？未保存的更改将丢失。')
  if (!answer) return false
})

// 路由参数变化时
onBeforeRouteUpdate((to, from) => {
  // 响应路由参数变化
  console.log('Route updated:', to.params)
})
</script>
```

## 动态路由

### 添加路由

```ts
router.addRoute({
  path: '/new',
  component: NewComponent
})

// 添加嵌套路由
router.addRoute('parent', {
  path: 'child',
  component: ChildComponent
})
```

### 删除路由

```ts
const removeRoute = router.addRoute({ path: '/temp', component: Temp })

// 删除路由
removeRoute()

// 或通过名称删除
router.removeRoute('temp')
```

### 检查路由

```ts
router.hasRoute('user')  // 检查路由是否存在

```

## 导航解析流程

```text
1. 导航被触发
2. 在失活的组件里调用 beforeRouteLeave 守卫
3. 调用全局的 beforeEach 守卫
4. 在重用的组件里调用 beforeRouteUpdate 守卫
5. 在路由配置里调用 beforeEnter
6. 解析异步路由组件
7. 在被激活的组件里调用 beforeRouteEnter
8. 调用全局的 beforeResolve 守卫
9. 导航被确认
10. 调用全局的 afterEach 钩子
11. 触发 DOM 更新
12. 调用 beforeRouteEnter 守卫中传给 next 的回调函数
```

## 最佳实践

### ✅ 使用命名路由

```vue
<!-- ✅ 推荐 -->
<RouterLink :to="{ name: 'user', params: { id: 123 } }">User</RouterLink>

<!-- ❌ 不推荐 -->
<RouterLink to="/user/123">User</RouterLink>
```

### ✅ 路由参数类型安全

```ts filename="router/typed-routes.ts"
import type { RouteRecordRaw } from 'vue-router'

export interface UserParams {
  id: string
}

export const userRoute: RouteRecordRaw = {
  path: '/user/:id',
  name: 'user',
  component: () => import('@/views/User.vue')
}
```

### ✅ 集中管理路由守卫

```ts filename="router/guards.ts"
import type { Router } from 'vue-router'

export function setupGuards(router: Router) {
  // 认证守卫
  router.beforeEach((to, from, next) => {
    if (to.meta.requiresAuth && !isAuthenticated()) {
      next({ name: 'login' })
    } else {
      next()
    }
  })
  
  // 权限守卫
  router.beforeEach((to, from, next) => {
    const roles = to.meta.roles as string[] | undefined
    if (roles && !hasRole(roles)) {
      next({ name: 'forbidden' })
    } else {
      next()
    }
  })
  
  // 页面标题
  router.afterEach((to) => {
    document.title = (to.meta.title as string) || 'App'
  })
}
```

```ts filename="router/index.ts"
import { setupGuards } from './guards'

const router = createRouter({ /* ... */ })

setupGuards(router)

export default router
```

### ❌ 避免在守卫中使用异步 setup

```vue
<!-- ❌ 错误 -->
<script setup lang="ts">
const data = await fetch('/api/data')  // 会阻塞路由导航
</script>

<!-- ✅ 正确 -->
<script setup lang="ts">
import { onMounted, ref } from 'vue'

const data = ref(null)

onMounted(async () => {
  data.value = await fetch('/api/data')
})
</script>
```
