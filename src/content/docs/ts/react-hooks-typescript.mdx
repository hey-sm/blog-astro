# React Hooks 类型

各类 React Hooks 在 TypeScript 中的类型标注方式，以及泛型 Hook 的自定义实现。

## useState

### 自动推断 vs 显式指定

```tsx
// ✅ 简单值——自动推断
const [count, setCount] = useState(0)
// count: number
// setCount: Dispatch<SetStateAction<number>>

const [name, setName] = useState('Alice')
// name: string

// ✅ 复杂类型——显式指定泛型
interface User {
  id: number
  name: string
  email: string
}

const [user, setUser] = useState<User | null>(null)

// ❌ 推断为 never[]，没有任何值可以插入
const [items, setItems] = useState([])
// ✅ 修复
const [items, setItems] = useState<string[]>([])

// ✅ 联合类型需要显式标注
const [status, setStatus] = useState<'idle' | 'loading' | 'success' | 'error'>('idle')
```

### SetStateAction 类型

```ts
// SetStateAction<T> = T | ((prev: T) => T)

// 所以 setState 可以接收新值或更新函数：
setUser({ id: 1, name: 'Alice', email: 'a@b.com' })  // ✅
setUser(prev => prev ? { ...prev, name: 'Bob' } : null)  // ✅ 类型安全
```

### useState 的初始化函数

```tsx
// ✅ 惰性初始化——不传泛型推断
const [data, setData] = useState(() => {
  const saved = localStorage.getItem('data')
  return saved ? (JSON.parse(saved) as User) : null
})
// data: User | null

// ✅ 带泛型的惰性初始化
const [data, setData] = useState<User | null>(() => {
  try {
    const saved = localStorage.getItem('user')
    return saved ? JSON.parse(saved) : null
  } catch {
    return null
  }
})
```

## useReducer

### 基础用法

```tsx
type State = { count: number }
type Action =
  | { type: 'increment' }
  | { type: 'decrement' }
  | { type: 'reset'; payload: number }

const reducer = (state: State, action: Action): State => {
  switch (action.type) {
    case 'increment':
      return { count: state.count + 1 }
    case 'decrement':
      return { count: state.count - 1 }
    case 'reset':
      return { count: action.payload }
  }
}

function Counter() {
  const [state, dispatch] = useReducer(reducer, { count: 0 })
  // state: State
  // dispatch: Dispatch<Action>

  return (
    <div>
      <p>{state.count}</p>
      <button onClick={() => dispatch({ type: 'increment' })}>+</button>
      <button onClick={() => dispatch({ type: 'reset', payload: 0 })}>Reset</button>
    </div>
  )
}
```

### 更复杂的 reducer

```tsx
interface Todo {
  id: number
  text: string
  completed: boolean
}

type TodoState = Todo[]

type TodoAction =
  | { type: 'ADD'; text: string }
  | { type: 'TOGGLE'; id: number }
  | { type: 'REMOVE'; id: number }

function todoReducer(state: TodoState, action: TodoAction): TodoState {
  switch (action.type) {
    case 'ADD':
      return [...state, { id: Date.now(), text: action.text, completed: false }]
    case 'TOGGLE':
      return state.map(todo =>
        todo.id === action.id ? { ...todo, completed: !todo.completed } : todo
      )
    case 'REMOVE':
      return state.filter(todo => todo.id !== action.id)
  }
}
```

## useRef

### DOM 元素 Ref

```tsx
function TextInput() {
  const inputRef = useRef<HTMLInputElement>(null)

  // inputRef.current: HTMLInputElement | null

  const focusInput = () => {
    inputRef.current?.focus()
  }

  return <input ref={inputRef} />
}
```

### 可变值 Ref（不是 DOM）

```tsx
// ✅ ref 用于存储可变值
const intervalRef = useRef<number>(0)
const timerRef = useRef<NodeJS.Timeout | null>(null)

function useInterval(callback: () => void, delay: number | null) {
  const callbackRef = useRef(callback)  // 不改类型则为 MutableRefObject

  useEffect(() => {
    callbackRef.current = callback
  }, [callback])

  useEffect(() => {
    if (delay === null) return
    const id = setInterval(() => callbackRef.current(), delay)
    return () => clearInterval(id)
  }, [delay])
}
```

### useRef 的只读 vs 可变

```tsx
// ✅ DOM Refs 是只读的（RefObject）——不能直接赋值
const divRef = useRef<HTMLDivElement>(null)
// divRef.current = document.createElement('div')  // ❌ 错误

// ✅ 可变值 Ref（MutableRefObject）——可以赋值
const valueRef = useRef<string>('initial')
valueRef.current = 'updated'  // ✅ 正确
```

## useContext

```tsx
// ✅ 定义 Context 类型
interface ThemeContextType {
  theme: 'light' | 'dark'
  toggleTheme: () => void
}

const ThemeContext = createContext<ThemeContextType | null>(null)

// ✅ 自定义 Hook 封装（推荐）
function useTheme() {
  const context = useContext(ThemeContext)
  if (!context) {
    throw new Error('useTheme must be used within ThemeProvider')
  }
  return context
}

// ✅ 使用
function ThemeToggle() {
  const { theme, toggleTheme } = useTheme()
  return (
    <button onClick={toggleTheme}>
      Current: {theme}
    </button>
  )
}
```

### createContext 的默认值

```tsx
// ✅ 提供有意义的默认值——避免 null
interface AuthContextType {
  user: User | null
  login: (email: string, password: string) => Promise<void>
  logout: () => void
}

const AuthContext = createContext<AuthContextType>({
  user: null,
  login: async () => {},
  logout: () => {}
})

// 使用时不需要 null 检查
function useAuth() {
  return useContext(AuthContext)
}
```

## useMemo / useCallback

```tsx
// ✅ useMemo 自动推断返回类型
const sortedList = useMemo(() => {
  return items.filter(item => item.active).sort()
}, [items])
// sortedList: 自动推断类型

// ✅ useCallback 保持函数类型
const handleClick = useCallback((id: number) => {
  setSelected(id)
}, [])
// handleClick: (id: number) => void
```

## useEffect / useLayoutEffect

```tsx
// ✅ Effect 的回调不能返回 Promise
useEffect(() => {
  // ✅ 可以定义 async 函数并在内部调用
  const loadData = async () => {
    const data = await fetch('/api/data')
    setData(data)
  }
  loadData()
}, [])

// ❌ 不能直接返回 async 函数
useEffect(async () => {  // ❌ 错误
  const data = await fetch('/api/data')
}, [])
```

## 自定义 Hook

### 泛型数据获取 Hook

```tsx
function useFetch<T>(url: string) {
  const [data, setData] = useState<T | null>(null)
  const [loading, setLoading] = useState(true)
  const [error, setError] = useState<Error | null>(null)

  useEffect(() => {
    setLoading(true)
    fetch(url)
      .then(r => r.json())
      .then(data => setData(data as T))
      .catch(setError)
      .finally(() => setLoading(false))
  }, [url])

  return { data, loading, error } as const
}

// 使用：T 推断为 User[]
const { data, loading } = useFetch<User[]>('/api/users')
```

### 泛型表单 Hook

```tsx
function useForm<T extends Record<string, unknown>>(initialValues: T) {
  const [values, setValues] = useState<T>(initialValues)
  const [errors, setErrors] = useState<Partial<Record<keyof T, string>>>({})

  const setField = <K extends keyof T>(name: K, value: T[K]) => {
    setValues(prev => ({ ...prev, [name]: value }))
  }

  const setError = <K extends keyof T>(name: K, error: string) => {
    setErrors(prev => ({ ...prev, [name]: error }))
  }

  return { values, errors, setField, setError } as const
}

// 使用
interface LoginForm {
  email: string
  password: string
}

function LoginPage() {
  const { values, setField } = useForm<LoginForm>({
    email: '',
    password: ''
  })

  setField('email', 'alice@example.com')  // ✅
  // setField('email', 42)  // ❌ 错误
}
```

### 带有返回类型的 as const

```tsx
// ✅ as const 保留元组类型
function useToggle(initial: boolean = false) {
  const [value, setValue] = useState(initial)
  const toggle = useCallback(() => setValue(v => !v), [])
  const setTrue = useCallback(() => setValue(true), [])
  const setFalse = useCallback(() => setValue(false), [])

  return [value, { toggle, setTrue, setFalse }] as const
}

const [isOn, { toggle }] = useToggle()
// isOn: boolean
// toggle: () => void
```

## 陷阱

### 1. useState 初始化空数组

```tsx
// ❌ 永远不要这样
const [items, setItems] = useState([])  // never[]

// ✅ 始终标注类型
const [items, setItems] = useState<Todo[]>([])
```

### 2. useRef 初始值与泛型不匹配

```tsx
// ❌ 错误：null 不是 number
const ref = useRef<number>(null)

// ✅ 使用联合类型
const ref = useRef<number | null>(null)
```

### 3. useEffect 依赖数组中遗漏函数

```tsx
// ❌ 每次 render 都创建新函数，导致 effect 重复执行
function Component() {
  const fetchData = async () => { /* ... */ }

  useEffect(() => {
    fetchData()
  }, [])  // ❌ lint 警告：fetchData 未在依赖中
}

// ✅ 使用 useCallback 稳定函数引用
function Component() {
  const fetchData = useCallback(async () => { /* ... */ }, [])

  useEffect(() => {
    fetchData()
  }, [fetchData])
}
```

## 最佳实践

### ✅ 自定义 Hook 尾部使用 as const

```tsx
// ✅ 保留元组类型而非推断为联合数组
function useCounter() {
  const [count, setCount] = useState(0)
  const inc = () => setCount(c => c + 1)
  const dec = () => setCount(c => c - 1)

  return [count, { inc, dec }] as const
}
```

### ✅ Context 配合自定义 Hook 做非空断言

```tsx
function useUser() {
  const ctx = useContext(UserContext)
  if (!ctx) throw new Error('useUser must be used within UserProvider')
  return ctx
}
```

### ✅ 复杂 state 优先使用 useReducer

```tsx
// ✅ 当多个 state 互相依赖时，useReducer 类型更清晰
const [state, dispatch] = useReducer(reducer, initialState)
```
