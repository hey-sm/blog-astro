# 样式方案

Vue 支持多种样式解决方案，从原生 CSS 到 UI 组件库。

## Scoped CSS

### 基础用法

```vue filename="components/Button.vue"
<template>
  <button class="btn">
    <slot />
  </button>
</template>

<style scoped>
.btn {
  padding: 8px 16px;
  background: #42b883;
  color: white;
  border: none;
  border-radius: 4px;
  cursor: pointer;
  transition: background 0.3s;
}

.btn:hover {
  background: #35a372;
}

.btn:active {
  background: #2d8f63;
}
</style>
```

### 深度选择器

修改子组件样式：

```vue
<template>
  <div class="wrapper">
    <ChildComponent />
  </div>
</template>

<style scoped>
/* ❌ 不生效 */
.child-class {
  color: red;
}

/* ✅ 使用 :deep() */
:deep(.child-class) {
  color: red;
}

/* ✅ 或使用 >>> (某些预处理器不支持) */
>>> .child-class {
  color: red;
}
</style>
```

### 插槽选择器

```vue
<template>
  <div class="container">
    <slot />
  </div>
</template>

<style scoped>
/* 修改插槽内容样式 */
:slotted(.slot-content) {
  color: green;
  font-weight: bold;
}

/* 修改特定标签 */
:slotted(p) {
  margin: 10px 0;
}
</style>
```

### 全局选择器

```vue
<style scoped>
/* 在 scoped 中定义全局样式 */
:global(.global-class) {
  font-size: 16px;
}

/* 混合使用 */
.local-class :global(.global-class) {
  color: blue;
}
</style>
```

## CSS Modules

### 基础用法

```vue filename="components/Card.vue"
<template>
  <div :class="$style.card">
    <h3 :class="$style.title">{{ title }}</h3>
    <p :class="$style.content">{{ content }}</p>
  </div>
</template>

<script setup lang="ts">
defineProps<{
  title: string
  content: string
}>()
</script>

<style module>
.card {
  padding: 20px;
  border: 1px solid #ddd;
  border-radius: 8px;
}

.title {
  margin: 0 0 10px;
  font-size: 18px;
  font-weight: bold;
}

.content {
  margin: 0;
  color: #666;
}
</style>
```

### 自定义模块名

```vue
<template>
  <div :class="classes.container">
    <p :class="classes.text">Hello</p>
  </div>
</template>

<style module="classes">
.container {
  padding: 20px;
}

.text {
  color: blue;
}
</style>
```

### 组合类名

```vue
<script setup lang="ts">
import { computed } from 'vue'

const props = defineProps<{
  variant: 'primary' | 'secondary'
  size: 'small' | 'large'
}>()

const buttonClass = computed(() => [
  $style.btn,
  $style[`btn-${props.variant}`],
  $style[`btn-${props.size}`]
])
</script>

<template>
  <button :class="buttonClass">
    <slot />
  </button>
</template>

<style module>
.btn {
  padding: 8px 16px;
  border: none;
  border-radius: 4px;
}

.btn-primary {
  background: #42b883;
  color: white;
}

.btn-secondary {
  background: #f0f0f0;
  color: #333;
}

.btn-small {
  font-size: 12px;
  padding: 4px 8px;
}

.btn-large {
  font-size: 18px;
  padding: 12px 24px;
}
</style>
```

## CSS 变量注入

在 `<style>` 中使用 `<script>` 的变量：

```vue filename="components/ThemedButton.vue"
<script setup lang="ts">
import { ref } from 'vue'

const theme = ref({
  color: '#42b883',
  bgColor: '#f0f0f0',
  fontSize: 16
})

function changeTheme() {
  theme.value = {
    color: '#ff6b6b',
    bgColor: '#ffe66d',
    fontSize: 18
  }
}
</script>

<template>
  <button @click="changeTheme">
    Themed Button
  </button>
</template>

<style scoped>
button {
  color: v-bind('theme.color');
  background: v-bind('theme.bgColor');
  font-size: v-bind('theme.fontSize + "px"');
  padding: 10px 20px;
  border: none;
  border-radius: 4px;
  cursor: pointer;
}
</style>
```

## Sass / SCSS

### 安装

```bash
npm install -D sass
```

### 使用

```vue filename="components/Card.vue"
<template>
  <div class="card">
    <div class="card-header">
      <h3 class="card-title">{{ title }}</h3>
    </div>
    <div class="card-body">
      <slot />
    </div>
    <div class="card-footer">
      <slot name="footer" />
    </div>
  </div>
</template>

<style lang="scss" scoped>
$border-color: #ddd;
$border-radius: 8px;
$spacing: 20px;

.card {
  border: 1px solid $border-color;
  border-radius: $border-radius;
  overflow: hidden;
  
  &-header {
    padding: $spacing;
    background: #f5f5f5;
    border-bottom: 1px solid $border-color;
  }
  
  &-title {
    margin: 0;
    font-size: 18px;
    font-weight: bold;
  }
  
  &-body {
    padding: $spacing;
  }
  
  &-footer {
    padding: $spacing;
    background: #f9f9f9;
    border-top: 1px solid $border-color;
  }
}
</style>
```

### 全局样式变量

```scss filename="assets/styles/variables.scss"
// 颜色
$primary-color: #42b883;
$secondary-color: #35495e;
$danger-color: #ff6b6b;
$warning-color: #ffa502;
$success-color: #26de81;

// 间距
$spacing-xs: 4px;
$spacing-sm: 8px;
$spacing-md: 16px;
$spacing-lg: 24px;
$spacing-xl: 32px;

// 断点
$breakpoint-sm: 576px;
$breakpoint-md: 768px;
$breakpoint-lg: 992px;
$breakpoint-xl: 1200px;
```

```ts filename="vite.config.ts"
import { defineConfig } from 'vite'
import vue from '@vitejs/plugin-vue'

export default defineConfig({
  plugins: [vue()],
  css: {
    preprocessorOptions: {
      scss: {
        additionalData: `@import "@/assets/styles/variables.scss";`
      }
    }
  }

```

```vue filename="components/Button.vue"

// 可以直接使用全局变量
.btn {
  background: $primary-color;
  padding: $spacing-md;
  
  @media (max-width: $breakpoint-md) {
    padding: $spacing-sm;
  }
}
</style>
```

## Element Plus

### 安装

```bash
npm install element-plus
```

### 完整引入

```ts filename="main.ts"
import { createApp } from 'vue'
import ElementPlus from 'element-plus'
import 'element-plus/dist/index.css'
import App from './App.vue'

const app = createApp(App)

app.use(ElementPlus)
app.mount('#app')
```

### 按需引入

```bash
npm install -D unplugin-vue-components unplugin-auto-import
```

```ts filename="vite.config.ts"
import { defineConfig } from 'vite'
import vue from '@vitejs/plugin-vue'
import AutoImport from 'unplugin-auto-import/vite'
import Components from 'unplugin-vue-components/vite'
import { ElementPlusResolver } from 'unplugin-vue-components/resolvers'

export default defineConfig({
  plugins: [
    vue(),
    AutoImport({
      resolvers: [ElementPlusResolver()],
    }),
    Components({
      resolvers: [ElementPlusResolver()],
    }),
  ],
})
```

### 使用

```vue filename="components/UserForm.vue"
<script setup lang="ts">
import { ref, reactive } from 'vue'
import { ElMessage } from 'element-plus'

const formData = reactive({
  name: '',
  email: '',
  age: 0
})

const rules = {
  name: [
    { required: true, message: '请输入姓名', trigger: 'blur' }
  ],
  email: [
    { required: true, message: '请输入邮箱', trigger: 'blur' },
    { type: 'email', message: '请输入正确的邮箱', trigger: 'blur' }
  ]
}

const formRef = ref()

async function handleSubmit() {
  await formRef.value.validate()
  
  ElMessage.success('提交成功')
}
</script>

<template>
  <el-form
    ref="formRef"
    :model="formData"
    :rules="rules"
    label-width="80px"
  >
    <el-form-item label="姓名" prop="name">
      <el-input v-model="formData.name" />
    </el-form-item>
    
    <el-form-item label="邮箱" prop="email">
      <el-input v-model="formData.email" type="email" />
    </el-form-item>
    
    <el-form-item label="年龄">
      <el-input-number v-model="formData.age" :min="0" :max="120" />
    </el-form-item>
    
    <el-form-item>
      <el-button type="primary" @click="handleSubmit">提交</el-button>
      <el-button @click="formRef.reset()">重置</el-button>
    </el-form-item>
  </el-form>
</template>
```

## 最佳实践

### ✅ 使用 CSS 变量实现主题

```vue
<script setup lang="ts">
import { ref } from 'vue'

const theme = ref<'light' | 'dark'>('light')

function toggleTheme() {
  theme.value = theme.value === 'light' ? 'dark' : 'light'
}
</script>

<template>
  <div :class="theme">
    <button @click="toggleTheme">Toggle Theme</button>
    <div class="content">Content</div>
  </div>
</template>

<style scoped>
.light {
  --bg-color: #ffffff;
  --text-color: #333333;
}

.dark {
  --bg-color: #1a1a1a;
  --text-color: #ffffff;
}

.content {
  background: var(--bg-color);
  color: var(--text-color);
  padding: 20px;
}
</style>
```

### ✅ 组织样式文件

```text
assets/
├── styles/
│   ├── variables.scss    # 变量
│   ├── mixins.scss       # 混入
│   ├── base.scss         # 基础样式
│   ├── utilities.scss    # 工具类
│   └── main.scss         # 主入口
```

### ❌ 避免过度使用 !important

```vue
<!-- ❌ 错误 -->
<style scoped>
.btn {
  color: red !important;
}
</style>

<!-- ✅ 正确：提高选择器优先级 -->
<style scoped>
.container .btn {
  color: red;
}
</style>
```
