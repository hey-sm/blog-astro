# React + TypeScript 基础

在 React 项目中使用 TypeScript，核心是掌握组件、Props、事件和常用 React 类型的标注方式。

## 配置 React + TypeScript 项目

```json filename="tsconfig.json"
{
  "compilerOptions": {
    "jsx": "react-jsx",          // React 17+ JSX Transform
    "lib": ["ES2022", "DOM"],
    "moduleResolution": "bundler",
    "strict": true,
    "esModuleInterop": true,
    "skipLibCheck": true
  }
}
```

## 函数组件 typed props

### 基本 Props

```tsx
interface UserCardProps {
  name: string
  email: string
  age: number
}

function UserCard({ name, email, age }: UserCardProps) {
  return (
    <div>
      <h3>{name}</h3>
      <p>{email}</p>
      <span>{age} years old</span>
    </div>
  )
}
```

### 带可选 children

```tsx
import type { PropsWithChildren } from 'react'

interface CardProps {
  title: string
  className?: string
}

// ✅ 方式 1：显式声明 children
function Card({ title, className, children }: PropsWithChildren<CardProps>) {
  return (
    <div className={className}>
      <h2>{title}</h2>
      {children}
    </div>
  )
}

// ✅ 方式 2：PropsWithChildren
function Card(props: PropsWithChildren<CardProps>) {
  // 等价
}
```

### React.FC 的使用与争议

```tsx
// ✅ 使用 React.FC（自动包含 children）
interface ButtonProps {
  label: string
  onClick: () => void
}

const Button: React.FC<ButtonProps> = ({ label, onClick }) => {
  return <button onClick={onClick}>{label}</button>
}

// ✅ 不推荐 React.FC（React 18 之后 children 不再是默认注入）
// 直接用普通函数 + 显式 Props
function Button({ label, onClick, children }: PropsWithChildren<ButtonProps>) {
  return <button onClick={onClick}>{children ?? label}</button>
}
```

## ReactNode vs ReactElement vs JSX.Element

```ts
import type { ReactNode, ReactElement } from 'react'

// ✅ ReactNode：最宽泛——组件可以渲染的任何内容
type Node = ReactNode  // string | number | boolean | null | undefined | ReactElement | ReactPortal | ...

// ✅ ReactElement：JSX 表达式的值
const element: ReactElement = <div>Hello</div>

// ✅ JSX.Element：等价于 ReactElement（全局可用）
const jsx: JSX.Element = <div>Hello</div>

// 使用建议
interface Props {
  // ✅ 想接受任何可渲染内容 → ReactNode
  children: ReactNode

  // ✅ 需要确认是具体的元素 → ReactElement
  header?: ReactElement
}
```

## 事件类型

```tsx
function Form() {
  // ✅ 表单事件
  const handleSubmit = (e: React.FormEvent<HTMLFormElement>) => {
    e.preventDefault()
  }

  // ✅ Input 变更事件
  const handleChange = (e: React.ChangeEvent<HTMLInputElement>) => {
    console.log(e.target.value)
  }

  // ✅ 按钮点击
  const handleClick = (e: React.MouseEvent<HTMLButtonElement>) => {
    console.log(e.clientX, e.clientY)
  }

  // ✅ 键盘事件
  const handleKeyDown = (e: React.KeyboardEvent<HTMLInputElement>) => {
    if (e.key === 'Enter') { /* ... */ }
  }

  // ✅ 焦点事件
  const handleFocus = (e: React.FocusEvent<HTMLInputElement>) => {
    console.log(e.target.value)
  }

  // ✅ 剪贴板事件
  const handlePaste = (e: React.ClipboardEvent<HTMLInputElement>) => {
    const text = e.clipboardData.getData('text')
  }

  return (
    <form onSubmit={handleSubmit}>
      <input
        onChange={handleChange}
        onKeyDown={handleKeyDown}
        onFocus={handleFocus}
        onPaste={handlePaste}
      />
      <button type="submit" onClick={handleClick}>
        Submit
      </button>
    </form>
  )
}
```

## 常用事件类型速查

| Prop 名称 | 类型 |
|-----------|------|
| `onChange` | `React.ChangeEvent<HTMLInputElement>` |
| `onSubmit` | `React.FormEvent<HTMLFormElement>` |
| `onClick` | `React.MouseEvent<HTMLButtonElement>` |
| `onKeyDown/Up` | `React.KeyboardEvent<HTMLInputElement>` |
| `onFocus` | `React.FocusEvent<HTMLInputElement>` |
| `onBlur` | `React.FocusEvent<HTMLInputElement>` |
| `onScroll` | `React.UIEvent<HTMLDivElement>` |
| `onDrag` | `React.DragEvent<HTMLDivElement>` |

## ref 类型

### DOM 元素的 ref

```tsx
function TextInput() {
  // ✅ ref 指向 HTMLInputElement
  const inputRef = useRef<HTMLInputElement>(null)

  const focus = () => {
    inputRef.current?.focus()
  }

  return <input ref={inputRef} />
}
```

### forwardRef

```tsx
// ✅ forwardRef 组件
interface InputProps {
  label: string
  error?: string
}

const Input = forwardRef<HTMLInputElement, InputProps>(
  ({ label, error, ...rest }, ref) => {
    return (
      <div>
        <label>{label}</label>
        <input ref={ref} {...rest} />
        {error && <span className="error">{error}</span>}
      </div>
    )
  }
)

Input.displayName = 'Input'
```

**React 19 `ref` as prop**：

```tsx
// React 19+ 可以直接通过 props.ref 访问 ref（推荐）
function Input({ label, ref, error, ...rest }: InputProps & { ref: React.Ref<HTMLInputElement> }) {
  return (
    <div>
      <label>{label}</label>
      <input ref={ref} {...rest} />
      {error && <span className="error'">{error}</span>}
    </div>
  )
}
```

### 命令式句柄（useImperativeHandle）

```tsx
interface ModalHandle {
  open: () => void
  close: () => void
}

const Modal = forwardRef<ModalHandle, { title: string; children: ReactNode }>(
  ({ title, children }, ref) => {
    const [isOpen, setIsOpen] = useState(false)

    useImperativeHandle(ref, () => ({
      open: () => setIsOpen(true),
      close: () => setIsOpen(false)
    }))

    if (!isOpen) return null

    return (
      <div className="modal-overlay">
        <div className="modal">
          <h2>{title}</h2>
          {children}
        </div>
      </div>
    )
  }
)
```

## React.ComponentProps 提取原生组件 Props

```tsx
// ✅ 提取原生元素的 Props
type DivProps = React.ComponentProps<'div'>
type InputProps = React.ComponentProps<'input'>
type ButtonProps = React.ComponentProps<'button'>

// ✅ 选择性提取 + 扩展
interface CustomInputProps extends Omit<React.ComponentProps<'input'>, 'size'> {
  label: string
  error?: string
}

function CustomInput({ label, error, ...inputProps }: CustomInputProps) {
  return (
    <div>
      <label>{label}</label>
      <input {...inputProps} />
      {error && <span>{error}</span>}
    </div>
  )
}
```

## 陷阱

### 1. 事件处理函数中 this 类型的丢失

```tsx
// ❌ 类组件中常见
class Form extends React.Component {
  handleSubmit(e: React.FormEvent) {  // 不绑定 this 会导致 this 为 undefined
    e.preventDefault()
  }
}

// ✅ 箭头函数自动绑定
class Form extends React.Component {
  handleSubmit = (e: React.FormEvent) => {
    e.preventDefault()
  }
}
```

### 2. React.forwardRef 选择泛型序

```tsx
// ⚠️ forwardRef<RefType, PropsType>——注意顺序是 ref 在前、props 在后
const Comp = forwardRef<HTMLInputElement, InputProps>((props, ref) => {})
```

### 3. JSX 中泛型组件需要 extends

```tsx
// ❌ 在 .tsx 中 <T> 会被误认为 JSX 标签
const Comp = <T>(props: T) => <div />

// ✅ 加逗号消除歧义
const Comp = <T,>(props: T) => <div />

// ✅ 或加 extends
const Comp = <T extends unknown>(props: T) => <div />
```

## 最佳实践

### ✅ Props 使用 interface（允许声明合并）

```tsx
interface ButtonProps {
  variant: 'primary' | 'secondary'
  size: 'sm' | 'md' | 'lg'
}
```

### ✅ 使用 React.ComponentProps 而不是手写原生元素 Props

```tsx
// ✅ 复用原生类型
type NativeProps = React.ComponentProps<'div'>
type CustomProps = { title: string } & Omit<NativeProps, 'children'>
```

### ✅ children 类型使用 ReactNode

```tsx
interface Props {
  children: ReactNode  // 最通用
}
```
