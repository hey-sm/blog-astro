# Props、Events、v-model

本章深入探讨 Vue 组件通信的三大机制。

## Props 深入

### 响应式解构（Vue 3.5+）

```vue
<script setup lang="ts">
// ✅ 保持响应式（Vue 3.5+）
const { name, age = 18 } = defineProps<{
  name: string
  age?: number
}>()
</script>

<template>
  <p>{{ name }} - {{ age }}</p>
</template>
```

### Boolean Props

```vue
<script setup lang="ts">
defineProps<{
  disabled?: boolean
}>()
</script>

<template>
  <!-- 以下都会传递 true -->
  <MyComponent disabled />
  <MyComponent :disabled="true" />
  <MyComponent disabled="true" />
</template>
```

## v-model 原理

### 基础 v-model

```vue
<!-- 语法糖 -->
<CustomInput v-model="text" />

<!-- 等价于 -->
<CustomInput
  :model-value="text"
  @update:model-value="text = $event"
/>
```

### 实现自定义 v-model

**方式 1：使用 defineModel（Vue 3.4+，推荐）**

```vue filename="components/CustomInput.vue"
<script setup lang="ts">
const model = defineModel<string>()
</script>

<template>
  <input v-model="model">
</template>
```

**方式 2：手动实现**

```vue filename="components/CustomInput.vue"
<script setup lang="ts">
const props = defineProps<{ modelValue: string }>()
const emit = defineEmits<{ 'update:modelValue': [value: string] }>()

function updateValue(e: Event) {
  emit('update:modelValue', (e.target as HTMLInputElement).value)
}
</script>

<template>
  <input :value="props.modelValue" @input="updateValue">
</template>
```

### 多个 v-model

```vue filename="components/UserForm.vue"
<script setup lang="ts">
const firstName = defineModel<string>('firstName')
const lastName = defineModel<string>('lastName')
</script>

<template>
  <input v-model="firstName" placeholder="First Name">
  <input v-model="lastName" placeholder="Last Name">
</template>
```

使用：

```vue
<script setup lang="ts">
import { ref } from 'vue'

const first = ref('')
const last = ref('')
</script>

<template>
  <UserForm v-model:first-name="first" v-model:last-name="last" />
</template>
```

### v-model 修饰符

```vue filename="components/CustomInput.vue"
<script setup lang="ts">
const [model, modifiers] = defineModel<string>({
  set(value) {
    if (modifiers.capitalize) {
      return value.charAt(0).toUpperCase() + value.slice(1)
    }
    return value
  }
})
</script>

<template>
  <input v-model="model">
</template>
```

使用：

```vue
<template>
  <CustomInput v-model.capitalize="text" />
</template>
```

## 事件系统深入

### 事件参数

```vue
<script setup lang="ts">
const emit = defineEmits<{
  // 无参数
  submit: []
  // 单个参数
  update: [value: string]
  // 多个参数
  change: [id: number, value: string, timestamp: number]
}>()

function handleSubmit() {
  emit('submit')
}

function handleChange(id: number, value: string) {
  emit('change', id, value, Date.now())
}
</script>
```

### 事件修饰符

```vue
<template>
  <!-- 父组件可以使用修饰符 -->
  <CustomButton @click.once="handler" />
  <CustomForm @submit.prevent="onSubmit" />
</template>
```

## 透传 Attributes 深入

### class 和 style 合并

```vue filename="components/Button.vue"
<template>
  <button class="btn">
    <slot />
  </button>
</template>
```

```vue filename="App.vue"
<template>
  <!-- class 会自动合并 -->
  <Button class="primary">Click</Button>
  <!-- 渲染为：<button class="btn primary">Click</button> -->
</template>
```

### 监听器继承

```vue filename="components/CustomInput.vue"
<template>
  <div class="wrapper">
    <!-- 事件监听器会透传 -->
    <input v-bind="$attrs">
  </div>
</template>
```

```vue
<template>
  <CustomInput @focus="handleFocus" @blur="handleBlur" />
</template>
```

## 最佳实践

### ✅ 使用 TypeScript 类型

```vue
<script setup lang="ts">
interface User {
  id: number
  name: string
}

const props = defineProps<{
  user: User
  users: User[]
}>()

const emit = defineEmits<{
  select: [user: User]
  delete: [id: number]
}>()
</script>
```

### ✅ Props 验证

```vue
<script setup lang="ts">
const props = defineProps({
  status: {
    type: String as PropType<'pending' | 'success' | 'error'>,
    required: true,
    validator: (value: string) => ['pending', 'success', 'error'].includes(value)
  }
})
</script>
```

### ❌ 避免直接修改 Props

```vue
<!-- ❌ 错误 -->
<script setup lang="ts">
const props = defineProps<{ count: number }>()

function increment() {
  props.count++  // ❌ 运行时警告
}
</script>

<!-- ✅ 正确：使用本地状态或事件 -->
<script setup lang="ts">
import { ref } from 'vue'

const props = defineProps<{ count: number }>()
const localCount = ref(props.count)

function increment() {
  localCount.value++
}
</script>
```
